;以下はeraBEMANIVer2.23時点における仕様

;DISABLE関数を使うとコマンド等が表示されなくなる 調教中コマンドとそれ以外で使い方が異なる
;コマンドの場合はCOMRESULTを0にする それ以外の場合は返り値に1を指定する
;IF文と組み合わせる事で条件を満たした場合は実行可能になるなどもできる
;★コマンド「スパンキング」をDISABLEする場合の構文例は下記
;	@DISABLE_スパンキング_（キャラ名）
;	COMRESULT = 0
;★MASTER側でコマンド「スパンキング」をDISABLEする場合の構文例は下記
;	@DISABLE_MASTER_スパンキング_（キャラ名）
;	COMRESULT = 0
;★娼館配属をDISABLEする場合の構文例は下記
;	@DISABLE_娼館配属_（キャラ名）
;	RETURN 1

;子供やクローンもDISABLEを使用できる
;★キャラの解体をそれぞれ通常キャラ、子供、クローンでDISABLEする場合の関数は下記
;	@DISABLE_EVENT_解体_（キャラ名）
;	@DISABLE_EVENT_解体_（父親キャラ名）×（母親キャラ名）
;	@DISABLE_EVENT_解体_CLONE_（複製元キャラ名）

;WIDEDISABLEは引数を参照する事で一部の関数のみは表示する事も可能
;一部関数のみ対応している場合や口上内で単純な文章表示に留まらない独自処理がある場合などの用途
;★構文例は下記 依代に憑依中調教開始と調教終了だけは表示して他は表示しない
	;@WIDEDISABLE_憑依時口上表示_（キャラ名）, 対象関数
	;#DIMS DYNAMIC 対象関数
	;SELECTCASE 対象関数
	;	CASE "KOJO_START", "KOJO_TURNEND"
	;		RETURN 0
	;	CASEELSE
	;		RETURN 1
	;ENDSELECT

@WIDEDISABLE_憑依時口上表示_（キャラ名）, 対象関数
#DIMS DYNAMIC 対象関数

@WIDEDISABLE_ボールギャグ時口上表示_（キャラ名）, 対象関数
#DIMS DYNAMIC 対象関数

	;憑依時や調教中にボールギャグをつけている時口上を自動的に非表示にする
	;SELECTCOMがボールギャグの場合つけていても表示される（外す時）
	;なおWIDEDISABLEが有効でもそれに関連する通常のDISABLEも判定されるため注意
	;ex.憑依時にKOJO_USE_ITEMを非表示にしても時の砂時計の使用自体をDISABLEされている場合使用不可能
	;☆対象関数の一覧（通常モードのほぼ全て）
	;	"KOJO_START"				：憑依時
	;	"KOJO_TURNEND"				：憑依時
	;	"KOJO_COM"					：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_SHOOT"		：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_TSHOOT"		：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_EX"			：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_AFTER"		：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_PALAM"		：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_MARK"			：憑依時、ボールギャグ
	;	"KOJO_MESSAGE_USERCOM"		：憑依時、ボールギャグ
	;	"KOJO_EVENT"				：憑依時
	;	"KOJO_USE_ITEM"				：憑依時
	;	"SPELLCAST"					：憑依時
	;	"RECIEVE_SPELLCAST"			：憑依時
	;	"KOJO_SPELLEND_（魔法名）"	：憑依時
	;★使用する場合の構文例は下記
		@WIDEDISABLE_憑依時口上表示_（キャラ名）, 対象関数
		#DIMS DYNAMIC 対象関数
			RETURN 1
		@WIDEDISABLE_ボールギャグ時口上表示_（キャラ名）, 対象関数
		#DIMS DYNAMIC 対象関数
			RETURN 1

;-------------------------------------------------
;  コマンド
;-------------------------------------------------
@DISABLE_（コマンド名）_（キャラ名）
	;指定したコマンドが表示されなくなる 関数内でCOMRESULTを直接0にする

@DISABLECOM_（キャラ名）, コマンド名
#DIMS DYNAMIC コマンド名
	;コマンドが表示されなくなる 単純に1を返すと全てのコマンドが表示されなくなるため注意
	;DISABLEしたいコマンドが多い時に関数一つで対応できる
	;TARGET、MASTER、ASSIいずれの場合でも呼ばれるためこのキャラがTARGETの場合のみ可能などの場合は別途分岐を作る必要がある
	;★構文例は下記
		SELECTCASE コマンド名
			CASE "正常位", "後背位", "騎乗位", "対面座位", "背面座位", "下着ずらし挿入", "乱れ牡丹"
				;充分濡れていない、または挿入側のペニスが汚れている場合実行不可
				SIF PALAM:潤滑 < PALAMLV:3 || GET_STAIN(STAIN:PLAYER:ペニス, "アナル", "尿")
					RETURN 1
		ENDSELECT

;  調教コマンド内選択肢
;-------------------------------------------------
@DISABLE_魔法詠唱選択_（キャラ名）
	;調教コマンドの魔法詠唱を魔法ごとにDISABLEするのに使う 返り値はビット管理
	;☆式中関数「魔法TOBIT」魔法名の文字列を数値に変換する
	;★構文例は下記
	;	#DIM DYNAMIC 禁止魔法
	;	SETBIT 禁止魔法, 魔法TOBIT("チェイン・オブ・ペイン")
	;	SETBIT 禁止魔法, 魔法TOBIT("終極オーバーキル")
	;	RETURN 禁止魔法

@DISABLE_プレゼント選択_（キャラ名）
	;プレゼントごとにDISABLEするのに使う 返り値はビット管理 特殊プレゼントは日付を条件に含めないと禁止できない
	;☆式中関数「プレゼントTOBIT」プレゼント名を数値に変換する
	;★構文例は下記
	;	#DIM DYNAMIC 禁止プレゼント
	;	SETBIT 禁止プレゼント, プレゼントTOBIT("特注品のディルド")
	;	RETURN 禁止プレゼント
